562 inline auto Plot::draw(std::string what, std::string use, std::string with) -> DrawSpecs&
563 {
564  // Save the draw arguments for this x,y data
565  m_drawspecs.emplace_back(what, use, with);
566 
567  // Set the default line style specification for this drawing (desired behavior is 1, 2, 3 (incrementing as new lines are plotted))
568  m_drawspecs.back().lineStyle(m_drawspecs.size());
569 
570  // Return the just created drawing object in case the user wants to customize it
571  return m_drawspecs.back();
572 }

574 template <typename X, typename... Vecs>
575 inline auto Plot::drawWithVecs(std::string with, const X& x, const Vecs&... vecs) -> DrawSpecs&
576 {
577  // Write the given vectors x and y as a new data set to the stream
578  std::ostringstream datastream;
579  gnuplot::writedataset(datastream, m_numdatasets, x, vecs...);
580 
581  // Set the using string to "" if X is not vector of strings.
582  // Otherwise, x contain xtics strings. Set the `using` string
583  // so that these are properly used as xtics.
584  std::string use;
585  if constexpr(internal::isStringVector<X>) {
586  const auto nvecs = sizeof...(Vecs);
587  use = "0:"; // here, column 0 means the pseudo column with numbers 0, 1, 2, 3...
588  for(auto i = 2; i <= nvecs + 1; ++i)
589  use += std::to_string(i) + ":"; // this constructs 0:2:3:4:
590  use += "xtic(1)"; // this terminates the string with 0:2:3:4:xtic(1), and thus column 1 is used for the xtics
591  }
592 
593  // Append new data set to existing data
594  m_data += datastream.str();
595 
596  // Draw the data saved using a data set with index `m_numdatasets`. Increase number of data sets
597  return draw("'" + m_datafilename + "' index " + internal::str(m_numdatasets++), use, with);
598 }

766 inline auto Plot::drawWithCols(std::string fname, std::string with, const std::vector<ColumnIndex>& cols) -> DrawSpecs&
767 {
768  std::string use;
769  for(auto col : cols)
770  use += col.value + ":"; // e.g., "1:4:5:7:" (where 1 is x, 4 is y, 5 is ylow and 7 is yhigh for a yerrorlines plot)
771  use = internal::trimright(use, ':'); // e.g., "1:4:5:7:" => "1:4:5:7"
772  std::string what = "'" + fname + "'"; // e.g., "'myfile.dat'"
773  return draw(what, use, with); // e.g., draw(what="'myfile.dat'", use="1:2", with="lines");
774 }

930 inline auto Plot::show() const -> void
931 {
932  // Open script file and truncate it
933  std::ofstream script(m_scriptfilename);
934 
935  // Add palette info. Use default palette if the user hasn't set one
936  gnuplot::palettecmd(script, m_palette.empty() ? internal::DEFAULT_PALETTE : m_palette);
937 
938  // Add terminal info
939  auto width = m_width == 0 ? internal::DEFAULT_FIGURE_WIDTH : m_width;
940  auto height = m_height == 0 ? internal::DEFAULT_FIGURE_HEIGHT : m_height;
941  std::string size = gnuplot::sizestr(width, height, false);
942  gnuplot::showterminalcmd(script, size, m_font);
943 
944  // Add the plot commands
945  script << repr();
946 
947  // Add an empty line at the end and close the script to avoid crashes with gnuplot
948  script << std::endl;
949  script.close();
950 
951  // save plot data to a file
952  savePlotData();
953 
954  // Show the plot
955  gnuplot::runscript(m_scriptfilename, true);
956 
957  // remove the temporary files if user wants to
958  if(m_autoclean)
959  {
960  cleanup();
961  }
962 }

964 inline auto Plot::save(std::string filename) const -> void
965 {
966  // Clean the file name to prevent errors
967  auto cleanedfilename = gnuplot::cleanpath(filename);
968 
969  // Get extension from file name
970  auto extension = cleanedfilename.substr(cleanedfilename.rfind(".") + 1);
971 
972  // Open script file
973  std::ofstream script(m_scriptfilename);
974 
975  // Add palette info. Use default palette if the user hasn't set one
976  gnuplot::palettecmd(script, m_palette.empty() ? internal::DEFAULT_PALETTE : m_palette);
977 
978  // Add terminal info
979  auto width = m_width == 0 ? internal::DEFAULT_FIGURE_WIDTH : m_width;
980  auto height = m_height == 0 ? internal::DEFAULT_FIGURE_HEIGHT : m_height;
981  std::string size = gnuplot::sizestr(width, height, extension == "pdf");
982  gnuplot::saveterminalcmd(script, extension, size, m_font);
983 
984  // Add output command
985  gnuplot::outputcmd(script, cleanedfilename);
986 
987  // Add the plot commands
988  script << repr();
989 
990  // Unset the output
991  script << std::endl;
992  script << "set output";
993 
994  // Add an empty line at the end and close the script to avoid crashes with gnuplot
995  script << std::endl;
996  script.close();
997 
998  // save plot data to a file
999  savePlotData();
1000 
1001  // Save the plot as a file
1002  gnuplot::runscript(m_scriptfilename, false);
1003 
1004  // remove the temporary files if user wants to
1005  if (m_autoclean)
1006  {
1007  cleanup();
1008  }
1009 }

1041  script << gnuplot::commandValueStr("set yrange", m_yrange);
1042  script << m_xlabel << std::endl;
1043  script << m_ylabel << std::endl;
1044  script << m_zlabel << std::endl;
1045  script << m_rlabel << std::endl;
1046  script << m_border << std::endl;
1047  script << m_grid << std::endl;
1048  script << m_style_fill << std::endl;
1049  script << m_style_histogram << std::endl;
1050  script << m_tics << std::endl;
1051  script << m_xtics_major_bottom << std::endl;
1052  script << m_xtics_major_top << std::endl;
1053  script << m_xtics_minor_bottom << std::endl;
1054  script << m_xtics_minor_top << std::endl;
1055  script << m_ytics_major_left << std::endl;
1056  script << m_ytics_major_right << std::endl;
1057  script << m_ytics_minor_left << std::endl;
1058  script << m_ytics_minor_right << std::endl;
1059  script << m_ztics_major << std::endl;
1060  script << m_ztics_minor << std::endl;
1061  script << m_rtics_major << std::endl;
1062  script << m_rtics_minor << std::endl;
1063  script << m_legend << std::endl;
1064  script << gnuplot::commandValueStr("set boxwidth", m_boxwidth);
1065  script << gnuplot::commandValueStr("set samples", m_samples);
1066 
1067  // Add custom gnuplot commands
1068  if (!m_customcmds.empty())
1069  {
1070  script << "#==============================================================================" << std::endl;
1071  script << "# CUSTOM EXPLICIT GNUPLOT COMMANDS" << std::endl;
1072  script << "#==============================================================================" << std::endl;
1073  for(const auto& c : m_customcmds)
1074  {
1075  script << c << std::endl;
1076  }
1077  }
1078 
1079  // Add the actual plot commands for all drawXYZ() calls
1080  script << "#==============================================================================" << std::endl;
1081  script << "# PLOT COMMANDS" << std::endl;
1082  script << "#==============================================================================" << std::endl;
1083  script << "plot \\\n"; // use `\` to have a plot command in each individual line!
1084 
1085  // Write plot commands and style per plot
1086  const auto n = m_drawspecs.size();
1087  for(std::size_t i = 0; i < n; ++i)
1088  script << " " << m_drawspecs[i] << (i < n - 1 ? ", \\\n" : ""); // consider indentation with 4 spaces!
1089 
1090  // Add an empty line at the end
1091  script << std::endl;
1092  return script.str();
