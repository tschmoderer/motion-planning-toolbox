1000  const OutputVector &fval,
1001  const JacobiMatrix &jacobian,
1002  const GradientVector &gradient,
1003  const Objective &objective,
1004  StepVector &step)
1005  {
1006  // approximate hessian
1007  const HessianMatrix hessian = jacobian.transpose() * jacobian;
1008 
1009  // precompute the full step length
1010  const StepVector fullStep = step;
1011  const auto fullStepLenSq = fullStep.squaredNorm();
1012 
1013  // compute the cauchy step
1014  const auto gradientLenSq = gradient.squaredNorm();
1015  const Scalar curvature = gradient.dot(hessian * gradient);
1016  const StepVector cauchyStep = (gradientLenSq / curvature) * gradient;
1017  const auto cauchyStepLenSq = cauchyStep.squaredNorm();
1018 
1019  // compute step diff
1020  const StepVector diffStep = fullStep - cauchyStep;
1021  const auto diffLenSq = diffStep.squaredNorm();
1022  const Scalar diffFac = cauchyStep.dot(diffStep) / diffLenSq;
1023 
1024  auto modelFitness = _param.acceptanceFitness() - Scalar{1};
1025  Index iteration = 0;
1026 
1027  // keep computing while the model fitness is bad
1028  while(modelFitness < _param.acceptanceFitness() &&
1029  (_param.maximumIterations() <= Index{0} || iteration < _param.maximumIterations()))
1030  {
1031  const auto radiusSq = _radius * _radius;
1032 
1033  // if the full step is within the trust region simply
1034  // use it, it provides a good minimizer
1035  if(fullStepLenSq <= radiusSq)
1036  {
1037  step = fullStep;
1038  }
1039  else
1040  {
1041  // if the cauchy step lies outside the trust region
1042  // go towards it until the trust region boundary
1043  if(cauchyStepLenSq >= radiusSq)
1044  {
1045  step = (_radius / std::sqrt(cauchyStepLenSq)) * cauchyStep;
1046  }
1047  else
1048  {
1049  const auto secondTerm = std::sqrt(diffFac * diffFac + (radiusSq + cauchyStepLenSq) / diffLenSq);
1050  const auto scale1 = -diffFac - secondTerm;
1051  const auto scale2 = -diffFac + secondTerm;
1052 
1053  step = cauchyStep + std::max(scale1, scale2) * (fullStep - cauchyStep);
1054  }
1055  }
1056 
1057  // compute the model fitness to determine the update scheme for
1058  // the trust region radius
1059  modelFitness = calulateModelFitness(xval, fval, gradient, hessian, step, objective);
1060 
1061  const auto stepLen = step.norm();
1062 
1063  // if the model fitness is really bad reduce the radius!
1064  if(modelFitness < static_cast<Scalar>(0.25))
1065  {
1066  _radius = static_cast<Scalar>(0.25) * stepLen;
1067  }
1068  // if the model fitness is very good then increase it
1069  else if(modelFitness > static_cast<Scalar>(0.75) && std::abs(stepLen - _radius) < _param.radiusEpsilon())
1070  {
1071  // use the double radius
1072  _radius = 2 * _radius;
1073  // maintain radius border if configured
1074  if(_param.maximumRadius() > 0 && _radius > _param.maximumRadius())
1075  _radius = _param.maximumRadius();
1076  }
1077 
1078  ++iteration;
1079  }
1080  }

1083  const Parameter &_param;
1084  Scalar _radius = Scalar{1};

1087  Scalar calulateModelFitness(const InputVector &xval,
1088  const OutputVector &fval,
1089  const GradientVector &gradient,
1090  const HessianMatrix &hessian,
1091  const StepVector &step,
1092  const Objective &objective) const
1093  {
1094  const Scalar error = fval.squaredNorm() / Scalar{2};
1095 
1096  // evaluate the error function at the new position
1097  InputVector xvalNext = xval + step;
1098  OutputVector fvalNext;
1099  JacobiMatrix jacobianNext;
1100  objective(xvalNext, fvalNext, jacobianNext);
1101  // compute the actual new error
1102  const Scalar nextError = fvalNext.squaredNorm() / Scalar{2};
1103  // compute the new error by the model
1104  const Scalar modelError = error + gradient.dot(step) + step.dot(hessian * step) / Scalar{2};
1105 
1106  return (error - nextError) / (error - modelError);
1107  }
