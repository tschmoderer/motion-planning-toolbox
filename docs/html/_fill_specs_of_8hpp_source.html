184 template <typename DerivedSpecs>
185 auto FillSpecsOf<DerivedSpecs>::repr() const -> std::string
186 {
187  std::string fillstyle; // ensure it remains empty if no fill style option has been given!
188  if(m_fillmode == "solid")
189  fillstyle = "fillstyle " + m_transparent + " solid " + m_density;
190  else if(m_fillmode == "pattern")
191  fillstyle = "fillstyle " + m_transparent + " pattern " + m_pattern_number;
192  else if(m_fillmode == "empty")
193  fillstyle = "fillstyle empty";
194 
195  std::string borderstyle; // ensure it remains empty if no border option has been given!
196  if(m_bordershow != "")
197  {
198  if(m_bordershow == "yes")
199  {
200  borderstyle = "border ";
201  borderstyle += gnuplot::optionValueStr("linecolor", m_bordercolor);
202  borderstyle += gnuplot::optionValueStr("linewidth", m_borderlinewidth);
203  }
204  else borderstyle = "noborder";
205  }
206 
207  std::stringstream ss;
208  ss << m_fillcolor << " " << fillstyle << " " << borderstyle;
209 
210  return internal::removeExtraWhitespaces(ss.str());
211 }
