157 inline auto FillStyleSpecs::repr() const -> std::string
158 {
159  std::string fillstyle; // ensure it remains empty if no fill style option has been given!
160  if(m_fillmode == "solid")
161  fillstyle = m_transparent + " solid " + m_density;
162  else if(m_fillmode == "pattern")
163  fillstyle = m_transparent + " pattern " + m_pattern_number;
164  else if(m_fillmode == "empty")
165  fillstyle = "empty";
166 
167  std::string borderstyle; // ensure it remains empty if no border option has been given!
168  if(m_bordershow != "")
169  {
170  if(m_bordershow == "yes")
171  {
172  borderstyle = "border ";
173  borderstyle += gnuplot::optionValueStr("linecolor", m_bordercolor);
174  borderstyle += gnuplot::optionValueStr("linewidth", m_borderlinewidth);
175  }
176  else borderstyle = "noborder";
177  }
178 
179  if(fillstyle.empty() && borderstyle.empty())
180  return "";
181 
182  std::string ss = "set style fill " + fillstyle + " " + borderstyle;
183 
184  return internal::removeExtraWhitespaces(ss);
185 }
