190 inline auto Figure::show() const -> void
191 {
192  // Open script file and truncate it
193  std::ofstream script(m_scriptfilename);
194 
195  // Add palette info. Use default palette if the user hasn't set one
196  gnuplot::palettecmd(script, m_palette.empty() ? internal::DEFAULT_PALETTE : m_palette);
197 
198  // Add terminal info
199  auto width = m_width == 0 ? internal::DEFAULT_FIGURE_WIDTH : m_width;
200  auto height = m_height == 0 ? internal::DEFAULT_FIGURE_HEIGHT : m_height;
201  std::string size = gnuplot::sizestr(width, height, false);
202  gnuplot::showterminalcmd(script, size, m_font);
203 
204  // Add multiplot commands
205  gnuplot::multiplotcmd(script, m_layoutrows, m_layoutcols, m_title);
206 
207  // Add the plot commands
208  for(const auto& row : m_plots)
209  for(const auto& plot : row)
210  script << plot.repr();
211 
212  // Add an empty line at the end and close the script to avoid crashes with gnuplot
213  script << std::endl;
214  script.close();
215 
216  // save plot data to file(s)
217  saveplotdata();
218 
219  // Show the figure
220  gnuplot::runscript(m_scriptfilename, true);
221 
222  // remove the temporary files if user wants to
223  if(m_autoclean)
224  {
225  cleanup();
226  }
227 }

229 inline auto Figure::save(const std::string& filename) const -> void
230 {
231  // Clean the file name to prevent errors
232  auto cleanedfilename = gnuplot::cleanpath(filename);
233 
234  // Get extension from file name
235  auto extension = cleanedfilename.substr(cleanedfilename.rfind(".") + 1);
236 
237  // Open script file
238  std::ofstream script(m_scriptfilename);
239 
240  // Add palette info. Use default palette if the user hasn't set one
241  gnuplot::palettecmd(script, m_palette.empty() ? internal::DEFAULT_PALETTE : m_palette);
242 
243  // Add terminal info
244  auto width = m_width == 0 ? internal::DEFAULT_FIGURE_WIDTH : m_width;
245  auto height = m_height == 0 ? internal::DEFAULT_FIGURE_HEIGHT : m_height;
246  std::string size = gnuplot::sizestr(width, height, extension == "pdf");
247  gnuplot::saveterminalcmd(script, extension, size, m_font);
248 
249  // Add output command
250  gnuplot::outputcmd(script, cleanedfilename);
251 
252  // Add multiplot commands
253  gnuplot::multiplotcmd(script, m_layoutrows, m_layoutcols, m_title);
254 
255  // Add the plot commands
256  for(const auto& row : m_plots)
257  for(const auto& plot : row)
258  script << plot.repr();
259 
260  // Close multiplot
261  script << "unset multiplot" << std::endl;
262 
263  // Unset the output
264  script << std::endl;
265  script << "set output";
266 
267  // Add an empty line at the end and close the script to avoid crashes with gnuplot
268  script << std::endl;
269  script.close();
270

272  saveplotdata();
273 
274  // Save the figure as a file
275  gnuplot::runscript(m_scriptfilename, false);
