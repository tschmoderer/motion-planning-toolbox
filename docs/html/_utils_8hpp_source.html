250 inline auto showterminalcmd(std::ostream& out, std::string size, std::string font) -> std::ostream&
251 {
252  out << "#==============================================================================" << std::endl;
253  out << "# TERMINAL" << std::endl;
254  out << "#==============================================================================" << std::endl;
255  // We set a terminal here to make sure we can also set a size. This is necessary, because the
256  // canvas size can ONLY be set using "set terminal <TERMINAL> size W, H".
257  // See: http://www.bersch.net/gnuplot-doc/canvas-size.html#set-term-size
258  // The GNUTERM variable contains the default terminal, which we're using for the show command.
259  // See: http://www.bersch.net/gnuplot-doc/unset.html
260  out << "set termoption enhanced" << std::endl;
261  if(font.size()) out << "set termoption " << font << std::endl;
262  out << "set terminal GNUTERM size " << size << std::endl;
263  return out;
264 }
